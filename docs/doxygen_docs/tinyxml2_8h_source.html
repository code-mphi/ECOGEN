365  virtual void* Alloc() {
366  if (!_root) {
367  // Need a new block.
368  Block* block = new Block();
369  _blockPtrs.Push(block);
370 
371  for (int i = 0; i<COUNT - 1; ++i) {
372  block->chunk[i].next = &block->chunk[i + 1];
373  }
374  block->chunk[COUNT - 1].next = 0;
375  _root = block->chunk;
376  }
377  void* result = _root;
378  _root = _root->next;
379 
380  ++_currentAllocs;
381  if (_currentAllocs > _maxAllocs) {
