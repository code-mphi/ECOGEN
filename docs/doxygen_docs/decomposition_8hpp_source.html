97  std::vector<key_type> initialize(int nProcs, int _rank, int restartSimulation) noexcept
98  {
99  std::vector<key_type> keys;
100 
101  if (restartSimulation == 0) {
102  auto nCells_t = std::accumulate(nCells_global_.begin(),nCells_global_.end(),1,
103  std::multiplies<int>());
104 
105  float chunks = static_cast<float>(nCells_t+0.5)/nProcs;
106  key_type key(0,0,0);
107  std::vector<int> nCells_per_rank;
108  for (int i = 0; i < nProcs; ++i)
109  {
110  size_t start = (size_t)(i*chunks);
111  size_t end = std::min(static_cast<int>((i+1)*chunks), nCells_t);
112  const int nlocal = end-start;
113  int count = 0;
114  key_rank_map_.emplace(key, i);
115  nCells_per_rank.emplace_back(nlocal);
116 
117  while (count < nlocal)
118  {
119  if (is_valid(key)) ++count;
120  ++key;
121  }
122 
123  //Store also end, to check validity:
124  if (i == nProcs-1)
125  {
126  nCells_per_rank.emplace_back(0);
127  key_rank_map_.emplace(--key, i);
128  }
129  }
130 
131  auto key_it = key_rank_map_.begin();
132  std::advance(key_it, _rank);
133  key = key_it->first;
134  auto nCells = nCells_per_rank[_rank];
135  keys.resize(nCells);
136 
137  for (int i = 0; i < nCells; ++i)
138  {
139  bool valid = false;

144  valid = true;

228  void communicateMaps(int _nCpu, std::vector<typename key_type::value_type>& localKeys, std::vector<int>& localRanks)
229  {

232

234  std::vector<int> mapSizes(_nCpu);

236 
237  //All gather maps (keys and ranks)
238  std::vector<int> displacements(_nCpu);
239  int sumMapSizes = 0;
240  for (int i = 0; i < _nCpu; ++i)
241  {
242  displacements[i] = sumMapSizes;
243  sumMapSizes += mapSizes[i];
244  }
245  std::vector<typename key_type::value_type> globalKeys(sumMapSizes);
246  std::vector<int> globalRanks(sumMapSizes);
247  MPI_Allgatherv(&localKeys[0], localMapSize, MPI_UNSIGNED_LONG_LONG, &globalKeys[0], &mapSizes[0], &displacements[0], MPI_UNSIGNED_LONG_LONG, MPI_COMM_WORLD);
248  MPI_Allgatherv(&localRanks[0], localMapSize, MPI_INT, &globalRanks[0], &mapSizes[0], &displacements[0], MPI_INT, MPI_COMM_WORLD);
249 
250  //Compose global map from keys and ranks
251  auto keyRankEnd = *(key_rank_map_.rbegin());
252  key_rank_map_.clear();
253  key_rank_map_.emplace(keyRankEnd);
254  for (std::size_t i = 0; i < globalKeys.size(); ++i)
255  {
256  key_rank_map_.emplace(globalKeys[i], globalRanks[i]);
257  }
